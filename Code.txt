#include <Servo.h>

// กำหนดขาเซนเซอร์
const int trigPin = 9;
const int echoPin = 10;

// สร้างอ็อบเจกต์เซอร์โว
Servo myServo;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  myServo.attach(6);
  myServo.write(0); // เริ่มที่ 0 องศา

  Serial.begin(9600);
}

void loop() {
  // ส่งสัญญาณ ultrasonic
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  // รับคลื่นสะท้อน
  long duration = pulseIn(echoPin, HIGH);
  float distance = duration * 0.034 / 2;

  // แสดงระยะทาง
  Serial.print("Distance: ");
  Serial.print(distance);
  Serial.println(" cm");

  // ตรวจจับวัตถุในระยะ 10 ซม.
  if (distance > 0 && distance <= 10.0) {
    Serial.println("Detected → Servo 90°");
    myServo.write(90);
  } else {
    Serial.println("Not Detected → Servo 0°");
    myServo.write(0);
  }

  delay(300);
}
