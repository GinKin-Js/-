#include <Servo.h>

#define TRIG_PIN 3
#define ECHO_PIN 4
#define MAX_DISTANCE_CM 20

Servo lid;

bool lidOpen = false;
unsigned long lidTimer = 0;

void setup() {
  lid.attach(A3);
  lid.write(0); // ปิดฝาตอนเริ่ม
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  Serial.begin(9600);
}

// ฟังก์ชันอ่านระยะจาก HC-SR04
int getDistanceCM() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  long duration = pulseIn(ECHO_PIN, HIGH, 20000); // รอไม่เกิน 20 ms
  int distance = duration * 0.034 / 2;
  return distance;
}

void loop() {
  int distance = getDistanceCM();
  unsigned long now = millis();

  // ถ้ามีมือเข้าใกล้ในระยะที่กำหนด → เปิดฝา
  if (distance > 1 && distance <= MAX_DISTANCE_CM) {
    if (!lidOpen) {
      lid.write(90); // เปิดฝา
      lidOpen = true;
      Serial.println("เปิดฝา");
    }
    lidTimer = now;
  }

  // ถ้าไม่มีมืออยู่แล้วเกิน 5 วินาที → ปิดฝา
  if (lidOpen && (now - lidTimer >= 5000)) {
    lid.write(0); // ปิดฝา
    lidOpen = false;
    Serial.println("ปิดฝา");
  }

  delay(100); // หน่วงเวลาเล็กน้อย
}
